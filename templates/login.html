const loginForm = document.getElementById('loginForm'); const errorMessage = document.getElementById('errorMessage'); loginForm.addEventListener('submit', async (e) => { e.preventDefault(); // Sayfanın yenilenmesini durdur const username = document.getElementById('username').value; const password = document.getElementById('password').value; try { const response = await fetch('/api/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username, password }) }); const data = await response.json(); if (response.ok && data.success) { // Backend'in beklediği formatta yönlendirme yapıyoruz window.location.href = `/dashboard/${data.department}?token=${data.token}`; } else { errorMessage.textContent = data.detail || 'Giriş başarısız!'; errorMessage.style.display = 'block'; } } catch (error) { console.error('Hata:', error); errorMessage.textContent = 'Sunucuya bağlanılamadı!'; errorMessage.style.display = 'block'; } });
